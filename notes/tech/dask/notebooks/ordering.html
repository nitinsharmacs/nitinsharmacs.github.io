    def evaluate(x1, y1, x2, y2):
    u = da.stack([x1, y1])
    v = da.stack([x2, y2])
    
    components = [u, v, u ** 2 + v ** 2]
    return [
        abs(c[0] - c[1]).mean(axis=-1)
        for c in components
    ]
